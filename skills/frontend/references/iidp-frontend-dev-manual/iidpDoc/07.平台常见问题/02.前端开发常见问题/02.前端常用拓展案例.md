## 拓展表格操作列的自定义按钮点击事件

> 需要先在 `grid.buttons` 上配置自定义按钮  
> **不推荐**直接拓展操作列，追加按钮并拓展点击事件: 点击该按钮时不会触发平台的同步数据逻辑，拿到的 row 数据可能会不正确

::: details 查看代码

```javascript
demo_operate_btn_xxx_extend_view: {
  selector: {
    attr: 'id',
    value: 'xxx_table_main_table' // id 选 xxx_table_main_table
  },
  type: 'custom',
  beforeOperate: (app, operateItem, options) => {
    if (!options.element._customRestClickFn) {
      //缓存当前的点击事件
      let clickFn = options.element.bind_on_clickOptionBtn;
      options.element.bind_on_clickOptionBtn = async (params) => {
        const { self: vm, value } = params
        // value 里可以取到 row 的值
        debugger
        // 判断是否点击的是自定义按钮 btn 是字段上配置的内容
        if (value?.btn?.xxx === 'xxx') {
        } else {
          // 其他按钮执行原本的逻辑
          clickFn(params);
        }
      }
      options.element._customRestClickFn = true; // 防止重复点击
    }
    return operateItem.view
  }
},
```

## 子表增加 openview 弹窗按钮，支持数据多选添加与状态回显

> 本案例聚焦于复杂的“多选+子表与主表一起保存”场景。  
> 如遇单选或子表单独保存，逻辑会相对简化，请根据实际需求参考并调整。

::: details 查看代码

```javascript
rbac_role_app_menu_tab_user_ids_table_toolbar_deleteEr_extend_view: {
    type: "before",
    selector: {
        attr: "id",
        value: "rbac_user_app_menu_tab_role_ids_table_toolbar_deleteEr",
    },
    view: {
        type: "button",
        value: "设置用户#",
        options: {
            type: "text",
            icon: "el-icon-circle-plus-outline",
            size: "medium",
        },
        view: {
            click: {
                openView: {
                    showType: "dialog",
                    preId: "rbac_user_dialog_content_",
                    model: "rbac_role",
                    type: "app_role_grid,app_role_search",
                    checkbox: "multiple",
                    labelField: "name",
                    valueField: "id",
                    confirm: (vm) => {
                        const dialogTable = vm.$select(vm.$ds.idPre + "table_main_table");
                        let subTable = vm.$select(
                            "rbac_user_app_menu_tab_role_ids_table_main_table"
                        );
                        // 新建 map 用于比对数据
                        dialogTable.$ds.selectListMap = new Map();
                        // 获取目前勾选的值
                        const checkedList = dialogTable?.$ds?.transferSelected || [];
                        checkedList.forEach((rawItem) => {
                            const item = rawItem; // 兼容数据源结构
                            dialogTable.$ds.selectListMap.set(item.id, rawItem);
                            // 如果初始数据中没有，说明是新增
                            if (!dialogTable?.$ds?.initTbDataMap?.get(item.id)) {
                                subTable?.$cmd.dealTableErCatch(
                                    "add",
                                    item,
                                    subTable,
                                    "tableData"
                                );
                            }
                        });

                        // 与打开弹窗时的初始数据计算删除
                        const initialData = dialogTable.$ds.initTbData || [];
                        initialData.forEach((initItem) => {
                            // 如果初始项不在当前选中 Map 中，说明被删除了
                            if (!dialogTable?.$ds.selectListMap.has(initItem.id)) {
                                subTable?.$cmd.dealTableErCatch(
                                    "delete",
                                    { id: initItem.id },
                                    subTable,
                                    "tableData"
                                );
                            }
                        });

                        // 清除缓存
                        dialogTable.$ds.initTbDataMap = new Map();
                    },
                    api: {
                        grid: {
                            search: {
                                reqPrep: async (vm, params) => {
                                    params.args.filter = [["roleType", "=", "0"]];
                                    return params;
                                },
                                reqAfter: async (vm, res) => {
                                    const dialogTable = vm.$select(
                                        vm.$ds.idPre + "table_main_table"
                                    );

                                    // --- 拦截判断：区分首次加载与分页/搜索 ---
                                    // 检查是否已经初始化过回显（非首次请求，如翻页或搜索不执行默认赋值逻辑）
                                    if (vm.$ds._hasInitDialogSelected) {
                                        return res;
                                    }
                                    // 标记为已初始化，防止后续翻页时覆盖用户的操作
                                    vm.$ds._hasInitDialogSelected = true;

                                    // --- 步骤1：获取全量已保存的网络数据 ---
                                    // 左侧勾选列表的数据需要展示全部，需要调接口（分页9999999）后在设置回显,回显需要处理成 { text: , value: } 的模式
                                    let searchParams = {
                                        params: {
                                            app: "sie-snest-tenant",
                                            model: "rbac_role",
                                            service: "search",
                                            args: {
                                                filter: [["user_ids", "=", "05pgjhl6wirh8"]],
                                                properties: ["name", "code", "remark"],
                                                limit: 999999999, // 不分页，获取全量
                                                offset: 0,
                                            },
                                        },
                                    };
                                    let selectedRes = await window.Tech.httpMeta({
                                        data: { ...searchParams },
                                    });

                                    const netList = Array.isArray(selectedRes?.data)
                                        ? selectedRes.data
                                        : selectedRes?.data?.data || [];

                                    // --- 步骤2：获取本地未保存的数据 ---
                                    // 获取父页面可能存在的本地数据(未入库的数据会暂时存留在表格的第一页，删除的数据存在缓存中)
                                    let subTableMain = tech_app.page.getNode(
                                        "rbac_user_app_menu_tab_role_ids_table_main_table"
                                    );
                                    let srcTbData = subTableMain?.data?.tableData || [];

                                    // --- 步骤3：获取本地已标记删除的数据 ---
                                    // 确定哪些 ID 已经被删除了，需要从最终列表中剔除
                                    const deletedIds = new Set();
                                    const tableErCache = subTableMain?.$ds
                                        ?.getRequest("tableData")
                                        ?.cache?.__erInstance?.get();
                                    if (tableErCache) {
                                        for (const [, [erType, targetId]] of tableErCache) {
                                            // erType 2 和 3 通常代表已删除状态
                                            if ((erType === 3 || erType === 2) && targetId)
                                                deletedIds.add(targetId);
                                        }
                                    }

                                    // --- 步骤4：合并数据并处理冲突 ---
                                    // 使用 Map 进行合并，以 ID 为键进行去重
                                    const dataMap = new Map();

                                    // 4.1 先存入网络数据作为基础
                                    netList.forEach(
                                        (item) => item.id && dataMap.set(item.id, item)
                                    );

                                    // 4.2 再存入本地数据
                                    // 如果 ID 相同，本地的 item 会直接覆盖网络数据，保证展示最新修改
                                    srcTbData.forEach(
                                        (item) => item.id && dataMap.set(item.id, item)
                                    );

                                    // 4.3 统一移除那些在本地被标记为删除的 ID
                                    deletedIds.forEach((id) => dataMap.delete(id));

                                    // --- 步骤5：格式化数据以适配组件要求 ---
                                    // 转换回数组，并补充穿梭框/左侧列表必须的 text 和 value 字段
                                    const finalData = Array.from(dataMap.values()).map(
                                        (item) => ({
                                            ...item,
                                            text: item.name,
                                            value: item.id,
                                        })
                                    );

                                    // --- 步骤6：设置底层回显依赖状态 ---
                                    // 同步数据，更新字段供确认按钮点击时比对新增和删除
                                    dialogTable.$ds.initTbData =
                                        window.Tech._cloneDeep(finalData);
                                    dialogTable.$ds.initTbDataMap = new Map();
                                    finalData.forEach(
                                        (item) =>
                                            item.id &&
                                            dialogTable.$ds.initTbDataMap.set(item.id, item)
                                    );

                                    // --- 步骤7：执行 UI 层面赋值回显 ---
                                    // 使用 setTimeout 确保在表格组件 DOM 渲染或内部状态准备好之后再赋值
                                    setTimeout(() => {
                                        // 右侧表格赋值：传入 ID 数组，表格会自动根据当前页数据匹配打钩
                                        dialogTable.data.defaultCheck = finalData.map(
                                            (item) => item.id
                                        );
                                        // 左侧勾选赋值：传入全量完整对象，左侧列表会全部渲染
                                        dialogTable.$ds.transferList = finalData;
                                        dialogTable.$ds.transferSelected = finalData;
                                    }, 200);

                                    return res;
                                },
                            },
                        },
                    },
                },
            },
        },
    },
},

```

### 方法二：手动拼接参数，回填到两个 form 内（ search_common 和 search_dropdown ）

::: details 查看代码

```javascript
const urlParams = vm.$select("custom_test_form")?.$ds?.form || {};
const commonForm = vm.$select(vm.$ds.idPre + "form_main_table_search_common");
const dropdownForm = vm.$select(
  vm.$ds.idPre + "form_main_table_search_dropdown",
);
const tableMainTable = vm.$select(vm.$ds.idPre + "table_main_table");
const tableSearch = vm.$select(vm.$ds.idPre + "form_main_table_search");
if (!tableSearch?.$cmd) {
  value.close();
  return;
}
const getItems = (form) =>
  form ? tableSearch.$cmd.getFormItems(form) || [] : [];
const totalItem = [...getItems(commonForm), ...getItems(dropdownForm)];

const tableHeaderItems =
  tableMainTable?.$ds?.tableHeaderParams?.tableHeaderItems || [];
tableHeaderItems.forEach((item) => {
  if (!totalItem.some((tItem) => tItem.name === item.name)) {
    totalItem.push(item);
  }
});

const headerFilterParams = tableMainTable?.$ds?.tableHeaderParams;
const formValues = {
  ...(commonForm?.$ds?.form || {}),
  ...(dropdownForm?.$ds?.form || {}),
};

const hookRes = await Tech.callHook(commonForm, "validateQuery", formValues, {
  searchTopForm: commonForm,
  searchBottomForm: dropdownForm,
  appendParams: headerFilterParams,
});

if (hookRes === false) {
  return;
}

const dropdownObj = vm.$select(vm.$ds.idPre + "dropdown_pop_form");
if (dropdownObj?.instance?.setOpen) {
  dropdownObj.instance.setOpen(false);
}

const cleanedRes = {};
Object.keys(hookRes || {}).forEach((key) => {
  if (key.indexOf("___values") === -1) {
    cleanedRes[key] = hookRes[key];
  }
});

const mergedParams = { ...cleanedRes, ...urlParams };

// 回填入参至表单
const filterArr =
  tableSearch.$cmd.meta.formFormat.handleSearchData(mergedParams, totalItem) ||
  [];
tableSearch.$cmd.initSearchFormData(mergedParams, commonForm, dropdownForm);
tableSearch.$ds.tableFilter = filterArr;
// 手动触发表格搜索
const searchBtn = vm.$select(idPre + "form_main_table_search_btn");
if (searchBtn) {
  setTimeout(() => {
    searchBtn?.instance?.click();
  }, 100);
}
```

## 表格列使用 transformValue 转换显示值

> `transformValue` 是表格列的值转换回调，用于将单元格原始值转换为显示文本。
>
> 签名：`(val, item, row) => {}`
>
> - `val`：单元格原始值（当前行数据中该字段的值）
> - `item`：当前列节点（表格节点 `items` 中的列节点，前端渲染后），**不是后端 `columns` 配置项**
> - `row`：当前行数据对象`

### 给已有列添加 transformValue

::: details 查看代码

```javascript
// 给表格已有的「状态」列动态添加 transformValue
demo_table_status_column_transform_value_extend: {
  type: 'custom',
  selector: {
    attr: 'id',
    value: 'menu_table_main_table',
  },
  beforeOperate: (app, operateItem, options) => {
    // 在表格节点的 items（列列表）中找到目标列
    const columns = options.element.items || [];
    const statusCol = columns.find(col => col.field === 'status');
    if (statusCol) {
      statusCol.transformValue = (val, item, row) => {
        if(!val || !Array.isArray(val) || val.length === 0) return val || ''
        try {
          const options = item.items || [];
          const found = options.find(opt => opt.value === val[0]);
          return found ? found.label : val;
        } catch (error) {
          console.error('transformValue error:', error);
          return val;
        }
      };
    }
    return operateItem.view;
  },
},
```

### 平台 grid 视图的 column 上配置使用

与 form 视图的 column 上配置使用类似，在 grid 视图内，需要使用 `v-model: model[value]` 双向绑定行数据

::: details 查看代码

```vue
<template>
  <div class="custom_table_vue_container">
    <!-- 点击文本打开弹窗 -->
    <span class="calc_text_display" @click="openDialog">
      {{ displayValue }}
    </span>
    <!-- 选择顺序的弹窗 -->
    <el-dialog :visible.sync="dialogVisible" title="请选择顺序" width="500px">
      <div class="calc_text_form">
        <!-- 单选组选项 -->
        <el-radio-group v-model="formSelected">
          <el-radio
            v-for="option in radioOptions"
            :key="option.value"
            :label="option.value"
          >
            {{ option.label }}
          </el-radio>
        </el-radio-group>
      </div>
      <span slot="footer" class="dialog-footer">
        <el-button @click="handleClear">清除</el-button>
        <el-button @click="dialogVisible = false">取消</el-button>
        <el-button type="primary" @click="handleConfirm">确定</el-button>
      </span>
    </el-dialog>
  </div>
</template>

<script>
export default {
  name: "tech-custom-table-vue",
  props: {
    model: {
      type: Object,
      required: true,
      default: () => ({}),
    },
    // 绑定到 model 的字段名
    name: {
      type: String,
      default: "",
    },
  },
  data() {
    return {
      // 控制弹窗显示
      dialogVisible: false,
      // 当前选择值
      formSelected: null,
      // 单选项配置
      radioOptions: [
        {
          value: 1,
          label: "第一",
        },
        {
          value: 2,
          label: "第二",
        },
        {
          value: 3,
          label: "第三",
        },
      ],
    };
  },
  computed: {
    // 文本显示：优先显示匹配到的 label
    displayValue() {
      const currentValue = this.model?.[this.name];
      if (
        currentValue === undefined ||
        currentValue === null ||
        currentValue === ""
      ) {
        return "请选择";
      }
      const matched = this.radioOptions.find(
        (option) =>
          String(option.value) === String(currentValue) ||
          option.label === currentValue,
      );
      return matched ? matched.label : currentValue;
    },
  },

  methods: {
    // 打开弹窗并回显当前值
    openDialog() {
      this.dialogVisible = true;
      const currentValue = this.model?.[this.name];
      const matched = this.radioOptions.find(
        (option) =>
          String(option.value) === String(currentValue) ||
          option.label === currentValue,
      );
      this.formSelected = matched ? matched.value : null;
    },
    // 确认选择并同步数据到外部表格组件 model
    handleConfirm() {
      this.$set(this.model, this.name, this.formSelected);
      this.$emit("change", this.formSelected);
      this.dialogVisible = false;
    },
    // 清空选择并同步数据到外部表格组件 model
    handleClear() {
      this.formSelected = null;
      this.$set(this.model, this.name, null);
      this.$emit("change", null);
    },
  },
};
</script>
```

```js
demo_watch_hearder_theme_data_change_extend_view: {
  type: 'custom',
  selector: {
    attr: 'id',
    value: 'dropdown_meta_header_skin'
  },
  beforeOperate: (app, operateItem, options) => {
    options.element.created = (vm) => {
      let skinNode = tech_app.page.getNode('dropdown_meta_header_skin');
      let skinDataReactive = window.Tech.reactiveFn(skinNode?.$ds);
      if (skinNode.data._demo_skin_Watch) {
        // 检查之前是否有对应watch 有则先销毁
        skinNode.data._demo_skin_Watch();
        skinNode.data._demo_skin_Watch = null;
      }
      // 创建watch
      skinNode.data._demo_skin_Watch = window.Tech.watchFn(
        () => skinDataReactive.skinText,
        (val, old) => {
          const themeCacheKey = window.Tech?.theme?.cacheKey || 'TECH-THEME';
          // 获取当前主题
          const themeDefine = JSON.parse(localStorage.getItem(themeCacheKey));
          // 执行需要动态触发的逻辑 - 此案例为动态增加样式
          if (themeDefine.value == 'CUSTOM-GREEN') {
            // 动态增加样式 - addClass
            tech_app.utils.addClass('green_theme_css', greenSkinCss );
          } else {
            tech_app.utils.removeClass('green_theme_css');
          }
        },
        { deep: true, immediate: true }
      );
    };
  }
}
```
